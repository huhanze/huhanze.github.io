1. Async概述

你可能会注意到在Vapor的一些APIs中返回了通用的 **

/// 假定我们从某写API获取到了一个 future string
let futureString: Future = …

/// 将future string 映射为一个 integer
let futureInt = futureString.map(to: Int.self) { string in
print(string) // The actual String
return Int(string) ?? 0
}

/// 现在我们拥有了一个future integer
print(futureInt) // Future

/// 假定我们从某写API获取到了一个 future string
let futureString: Future = …

/// 假定我们已经创建了一个HTTP客户端
let client: Client = …

/// 将future string映射为future response
let futureResponse = futureString.flatMap(to: Response.self) { string in
return client.get(string) // Future
}

/// 输出Future
print(futureResponse) // Future

/// 假定从一些API获取一个 void future
let userDidSave: Future = …

/// 将void future 转换为一个HTTP状态值
let futureStatus = userDidSave.transform(to: HTTPStatus.ok)
print(futureStatus) // Future

/// 假定我们从某写API获取到了一个 future string
let futureString: Future = …

/// 假定我们已经创建了一个HTTP客户端
let client: Client = …

/// 先将string转换为URL，在将其结果转换为 response
let futureResponse = futureString.map(to: URL.self) { string in
guard let url = URL(string: string) else {
throw Abort(.badRequest, reason: “Invalid URL string: (string)”)
}
return url
}.flatMap(to: Response.self) { url in
return client.get(url)
}

print(futureResponse) // Future

/// future string
let futureString: Future = …

futureString.do { string in
print(string) // 实际的string
}.catch { error in
print(error) // 捕获错误
}

let futureString: Future = …

futureString.always {
print(“The future is complete!”)
}

let futureString: Future = …

/// 在string处于就绪前会一直阻塞
let string = try futureString.wait()
print(string) /// String

/// 创建一个新的Promise
let promiseString = req.eventLoop.newPromise(String.self)
print(promiseString) // Promise
print(promiseString.futureResult) // Future

/// 成功关联future
promiseString.succeed(result: “Hello”)

/// 关联future失败
promiseString.fail(error: …)

print(app.eventLoop) // EventLoop

router.get(“hello”) { req in
/// 让当前的even loop的线程睡5秒
sleep(5)

/// 一旦线程被唤醒，返回Hello, world!字符串
return "Hello, world!"

}

router.get(“hello”) { req in
/// 创建一个void类型的 promise
let promise = req.eventLoop.newPromise(Void.self)

/// 将任务分添加到全局队列(分配在后台线程)
DispatchQueue.global() {
    /// 在后台线程睡5秒
    /// 这将不影响任何even-loop
    sleep(5)

    /// 当阻塞工作完成
    /// 完成promise及其关联的future
    promise.succeed()
}

/// 等待future以待完成, 
/// 然后转换结果
return promise.futureResult.transform(to: "Hello, world!")

}
```

并非所有的阻塞调用如sleep(_:)那么明显，如果你怀疑你使用的调用可能会发生阻塞，那么你需要查看方法的内部或者寻求他人帮助，如果函数正在执行磁盘操作或网络IO操作并使用同步API(没有回调或期futures)，那么很可能会阻塞。

提示：

如果阻塞工作是你应用程序的核心部分，您应该考虑使BlockingIOThreadPool来控制创建来执行阻塞工作的线程数量,这将帮助您避免在执行阻塞工作时占用CPU时间的even loop。
